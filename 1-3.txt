void tauschen (int& a, int &b)
{
    int help = a;
    a = b;
    b = help;
}
void tauschen (float& a, float &b)
{
    float help = a;
    a = b;
    b = help;
}
void tauschen(float a[], float  b[], int k)
{
    float help = a[k];
    a[k] = b[k];
    b[k] = help;
}
